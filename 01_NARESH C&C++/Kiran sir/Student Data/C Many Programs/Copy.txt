#include<stdio.h>
#include<conio.h>
void main(int argc,char *argv[])
{
FILE *fs,*ft;
char ch;
if(argc!=3)
{
printf("Insufficient arg.");
exit(1);
}
fs=fopen(argv[1],"r");
ft=fopen(argv[2],"w");
if(fs==NULL)
{
printf("cannot open the source file");
exit(1);
}
if(ft==NULL)
{
printf("cannot open the target file");
exit(1);
}
while(1)
{
ch=fgetc(fs);
if(ch==EOF)
break;
fputc(ch,ft);
}
fclose(fs);
fclose(ft);
}
