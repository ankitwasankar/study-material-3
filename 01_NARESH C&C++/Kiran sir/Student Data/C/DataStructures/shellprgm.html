   void shellsort(int elements[], int maxsize)
   {
   int i,j,increment,temp;
   increment=7;
   while(increment > 0)
   {
   for(i=0;i<maxsize;i++)
   {
   j=i;
   temp=elements[i];
   while((j>=increment)&&(elements[j-increment]>temp))
   elements[j]=elements[j-increment];
   j=j-increment;
   }
   elements[j]=temp;
   }
   if(increment/3!=0)
   increment=increment/3;
   else if(increment==1)
   increment=0;
   else
   increment=1;
   }
   }
