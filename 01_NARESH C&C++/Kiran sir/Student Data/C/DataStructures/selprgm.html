C Program using Selection Sort :

#include <stdio.h>
main()
{
int a[20],n,i;
printf(" How many numbers");
scanf(" %d",&n);
printf("Enter %d numbers",n);
for(i=0;i<n;++i)
scanf("%d",%a[i]);
printf(" Numbers before sort :");
for(i=0;i<n;++i)
printf("%d",a[i]);
SelectionSort(a,n);
printf("Numbers after sort:");
for(i=0;i<n;++i)
printf("%d",a[i]);
}
SelectionSort(int a[],int n)
{
int index, minposition,t,j;
for(index=0;index<n-1;index++)
{
minposition=index;
for(j=index+1;j<n;index++)
{
if(a[j]<a[minposition])
minposition=j;
}
t=a[index];
a[index]=a[minposition];
a[minposition]=t;
}
}
