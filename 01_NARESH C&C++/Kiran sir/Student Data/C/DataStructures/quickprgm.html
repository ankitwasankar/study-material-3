# include < stdio.h >

scanf("5d",&a[i]);
printf("NUmbers before sort");
for(i=0;i<n;++i)

for(i=0;i<n;++i)
printf("%d",a[i]);
}
qsort(int a[],int lb,int ub)
{
int i,j,pivot;
if(lb<ub)
{
i=lb;
j=ub;
pivot=lb;
while(i<j)
{
while((a[i]<=a[pivot])&&(i<ub))
i++;
while((a[j]>a[pivot])
j--;
if(i<j)
swap(&a[i],&a[j]);
}
swap(a[pivot],&a[i]);
qsort(a,lb,a[j-1]);
qsort(a,j+1,ub);
}
}
swap(int *a,int *b)
{
int t;
t=*a;
*a=*b;
*b=t;
}
